# =========================
# IRR stability / correctness
# =========================

# 基本: 単純な1回投資→1回回収
fin_irr(-100,110) => 0.1
fin_irr(-100,100) => 0
fin_irr(-100,90) => -0.1

# 既存の代表ケース
fin_irr(-100,30,40,50) => 0.0889633946933

# 解析的に分かるケース
# -100 + 121/(1+r)^2 = 0  => r = 0.1
fin_irr(-100,0,121) => 0.1

# -100 + 81/(1+r)^2 = 0   => r = -0.1
fin_irr(-100,0,81) => -0.1

# -100 + 133.1/(1+r)^3 = 0 => r = 0.1
fin_irr(-100,0,0,133.1) => 0.1

# 小さい正IRR
fin_irr(-100,101) => 0.01

# 小さい負IRR
fin_irr(-100,99) => -0.01

# ゼロ金利近傍
fin_irr(-1000,1000.000001) => 1e-9
fin_irr(-1000,999.999999) => -1e-9

# スケール不変性
fin_irr(-1,1.1) => 0.1
fin_irr(-1000000,1100000) => 0.1
fin_irr(-1e12,1.1e12) => 0.1

# 期数が長いケース
# -100 + 100*(1.01)^10 / (1+r)^10 = 0 => r = 0.01
fin_irr(-100,0,0,0,0,0,0,0,0,0,110.462212541) => 0.01

# 根が存在しない: 全部同符号
fin_irr(100,200,300) => Error
fin_irr(-100,-50,-10) => Error



# ほぼ -1 に近い根
# -100 + 0.01/(1+r) = 0 => 1+r = 0.0001 => r = -0.9999
fin_irr(-100,0.01) => -0.9999

# -1 近傍だが少し離れたケース
# -100 + 1/(1+r) = 0 => r = -0.99
fin_irr(-100,1) => -0.99

# 複数符号変化あり / 複数根候補のある系
# fin_irr(-100,230,-132) => skip

# 極端なキャッシュフローでの安定性
fin_irr(-1e-6,1.1e-6) => 0.1
fin_irr(-1e6,1.01e6) => 0.01

# IRRとNPVの整合確認用に作りやすいケース
# -100 + 50/(1+r) + 60/(1+r)^2 = 0
# 解は約 0.0639410298049853
fin_irr(-100,50,60) => 0.063941029805

# 長め・単調減少で収束しやすいケース
fin_irr(-1000,100,150,200,250,300,350) => 0.0768326354482

# rate≈-1 で bracket 検出の限界を踏むケース
# -100 + 0.001/(1+r) = 0 => r = -0.99999 は探索下限 start=-0.9999 より外
# 現実装では Error が正しい
fin_irr(-100,0.001) => Error
